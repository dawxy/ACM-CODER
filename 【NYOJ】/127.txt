int pow_mod(int a, int n)//ตน้
{
	if(!n) return 1;
	int x = pow_mod(a,n>>1);
	long long ans = (long long)x*x%MOD;
	if(n&1) ans = ans*a%MOD;
	return (int)ans;
}
